def calculate_archetype_fit(user_scores, archetype_profile):
    """
    Calculate how well user scores match an archetype
    Returns percentage match (0-100)
    """
    
    # Archetype profile defines ideal scores
    # Example for Strategic Planner:
    # {structure: 80, motivation: 75, focus: 80, task: 85}
    
    total_distance = 0
    for axis in ['structure', 'motivation', 'focus', 'task']:
        distance = abs(user_scores[axis] - archetype_profile[axis])
        total_distance += distance
    
    # Max possible distance = 400 (100 per axis × 4)
    # Convert to similarity percentage
    similarity = 100 - (total_distance / 4)
    
    return similarity
