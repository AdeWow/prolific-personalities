### Quiz Scoring Logic

**How to Calculate:**

```javascript
// Pseudocode
const calculateArchetype = (responses) => {
  let scores = {
    structure: 0,
    motivation: 0,
    cognitive: 0,
    task: 0
  };

  // Weight each response based on question mapping
  responses.forEach(response => {
    scores.structure += response.structureWeight;
    scores.motivation += response.motivationWeight;
    scores.cognitive += response.cognitiveWeight;
    scores.task += response.taskWeight;
  });

  // Normalize scores (0-100 scale)
  Object.keys(scores).forEach(key => {
    scores[key] = (scores[key] / maxPossibleScore) * 100;
  });

  // Find best-matching archetype
  return findClosestArchetype(scores);
};

const findClosestArchetype = (userScores) => {
  let bestMatch = null;
  let smallestDistance = Infinity;

  archetypes.forEach(archetype => {
    const distance = calculateEuclideanDistance(
      userScores,
      archetype.scoreRanges
    );
    if (distance < smallestDistance) {
      smallestDistance = distance;
      bestMatch = archetype;
    }
  });

  return bestMatch;
};
```

### Recommendation Algorithm

**Step 1: Calculate Fit Score**

```javascript
const calculateToolFit = (userScores, tool) => {
  // User's archetype fit
  const archetypeFit = tool.archetypeFit[userArchetype];
  
  // Axis alignment score
  const structureMatch = Math.abs(userScores.structure - tool.axisAlignment.structure) < 30 ? 1 : 0.5;
  const motivationMatch = tool.axisAlignment.motivation === 'both' ? 1 : 
                         (tool.axisAlignment.motivation === userMotivationStyle ? 1 : 0.5);
  
  // Combine scores
  const fitScore = (archetypeFit * 0.6) + 
                   (structureMatch * 0.2) + 
                   (motivationMatch * 0.2);
  
  return fitScore * 100; // Return 0-100 score
};
```

**Step 2: Filter & Sort**

```javascript
const recommendTools = (userProfile, filters = {}) => {
  let tools = toolDatabase;
  
  // Apply filters
  if (filters.maxPrice) {
    tools = tools.filter(t => t.pricing.free || t.pricing.paidPlans[0].price <= filters.maxPrice);
  }
  if (filters.category) {
    tools = tools.filter(t => t.category === filters.category);
  }
  if (filters.platform) {
    tools = tools.filter(t => t.platforms.includes(filters.platform));
  }
  
  // Calculate fit for each tool
  const scoredTools = tools.map(tool => ({
    ...tool,
    fitScore: calculateToolFit(userProfile, tool)
  }));
  
  // Sort by fit score
  return scoredTools.sort((a, b) => b.fitScore - a.fitScore).slice(0, 20);
};
```

**AI Chat Component:**

```tsx
'use client';

import { useState } from 'react';
import { Button } from '@/components/ui/button';
import { Input } from '@/components/ui/input';

export function AICoach({ archetype }: { archetype: string }) {
  const [messages, setMessages] = useState([
    { role: 'assistant', content: 'Hi! I can help you understand your results and create an action plan. What would you like to know?' }
  ]);
  const [input, setInput] = useState('');
  const [loading, setLoading] = useState(false);

  const sendMessage = async () => {
    if (!input.trim()) return;
    
    setMessages(prev => [...prev, { role: 'user', content: input }]);
    setInput('');
    setLoading(true);

    const response = await fetch('/api/chat', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ 
        message: input, 
        archetype,
        history: messages 
      })
    });

    const data = await response.json();
    setMessages(prev => [...prev, { role: 'assistant', content: data.response }]);
    setLoading(false);
  };

  return (
    <div className="bg-white rounded-lg shadow-lg p-6">
      <div className="mb-4 h-96 overflow-y-auto">
        {messages.map((msg, i) => (
          <div 
            key={i} 
            className={`mb-4 ${msg.role === 'user' ? 'text-right' : 'text-left'}`}
          >
            <div className={`inline-block p-3 rounded-lg ${
              msg.role === 'user' 
                ? 'bg-blue-500 text-white' 
                : 'bg-gray-100 text-gray-800'
            }`}>
              {msg.content}
            </div>
          </div>
        ))}
        {loading && <div>Agent is thinking...</div>}
      </div>
      
      <div className="flex gap-2">
        <Input 
          value={input}
          onChange={(e) => setInput(e.target.value)}
          onKeyPress={(e) => e.key === 'Enter' && sendMessage()}
          placeholder="Ask me anything..."
        />
        <Button onClick={sendMessage} disabled={loading}>
          Send
        </Button>
      </div>
    </div>
  );
}
```

**API Route for AI Agent:**

```typescript
// app/api/chat/route.ts
import { NextRequest, NextResponse } from 'next/server';
import Anthropic from '@anthropic-ai/sdk';

const anthropic = new Anthropic({
  apiKey: process.env.ANTHROPIC_API_KEY!,
});

export async function POST(req: NextRequest) {
  const { message, archetype, history } = await req.json();

  const systemPrompt = `You are a productivity coach specializing in the ${archetype} archetype. 
  
  Based on behavioral psychology research, provide personalized advice that aligns with this archetype's:
  - Natural working style
  - Motivation triggers
  - Focus patterns
  - Common blockers
  
  Be encouraging, specific, and research-backed. Cite relevant studies when helpful.`;

  const response = await anthropic.messages.create({
    model: 'claude-3-5-sonnet-20241022',
    max_tokens: 1024,
    system: systemPrompt,
    messages: [
      ...history.map((msg: any) => ({
        role: msg.role === 'user' ? 'user' : 'assistant',
        content: msg.content
      })),
      { role: 'user', content: message }
    ]
  });

  return NextResponse.json({ 
    response: response.content[0].text 
  });
}
```
