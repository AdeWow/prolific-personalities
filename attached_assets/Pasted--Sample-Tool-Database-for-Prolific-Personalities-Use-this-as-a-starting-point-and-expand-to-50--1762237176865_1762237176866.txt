// Export helper functions
export const getToolsByArchetype = (archetype: string): ProductivityTool[] => {
  return toolDatabase
    .map(tool => ({
      ...tool,
      fitScore: tool.archetypeFit[archetype as keyof typeof tool.archetypeFit]
    }))
    .sort((a, b) => b.fitScore - a.fitScore)
    .slice(0, 10);
};

export const getToolsByCategory = (category: string): ProductivityTool[] => {
  return toolDatabase.filter(tool => tool.category === category);
};

export const getToolById = (id: string): ProductivityTool | undefined => {
  return toolDatabase.find(tool => tool.id === id);
};
